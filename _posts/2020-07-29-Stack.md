#### 1. 스택

- 특정 위치에서만 원소 추가/제거가 가능함.

1. 원소 추가O(1)
2. 원소 제거O(1)
3. 최상단 원소 확인O(1)

구현은 배열과 특정 위치인 pos로 구현함

```c++
#include <bits/stdc++.h>

using namespace std;

const int MX = 100005;
int dat[MX];
int pos;

void push(int x) {
	dat[pos++] = x;
}
int pop() {
	int x;
	x = dat[--pos];
	return x;
}
int top() {
	if (pos-1) return dat[pos - 1];
}

```

#### 연습문제

1. 10828번 스택

```c++
//without STL

#include <bits/stdc++.h>

using namespace std;

const int MX = 100005;

int dat[MX];
int pos;

int main() {
	ios::sync_with_stdio(0);
	cin.tie();
	int N;
	cin >> N;
	string cmd;
	while (N--) {
		cin >> cmd;
		int num;
		if (cmd =="push") {
			cin >> num;

			dat[pos++] = num;
		}
		else if (cmd == "pop") {
			if (pos) {
				cout << dat[pos - 1] << "\n";
				pos--;
			}
			else
				cout << -1 << "\n";
		}
		else if (cmd == "top") {
			if (pos)
				cout << dat[pos-1] << "\n";
			else
				cout << -1 << "\n";
		}
		else if (cmd == "size") {
			cout << pos << "\n";
		}
		else if (cmd == "empty") {
			if (pos)
				cout << 0 << "\n";
			else
				cout << 1 << "\n";
		}
	}
	return 0;
}
```
