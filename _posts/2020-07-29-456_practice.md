```c++
#include <bits/stdc++.h>

using namespace std;

const int MX = 1000005;
int dat[MX];
int pos;

int main() {
	ios::sync_with_stdio(0);

	int N;
	int ret = 0;
	cin >> N;

	while (N--) {
		int k;
		cin >> k;
		if (k == 0) 
			pos--;
		else 
			dat[pos++] = k;

	}
	for (int i = 0; i < pos; i++) ret += dat[i];

	cout << ret;
}
```

#### leetcode. 232 Implement Queue using Stacks

2개의 스택을 이용해서 푼다. 문제 제출하면 컴파일에러가 왜 나는지는 모르겠다.

##### 1. Costlier pop (pop - O(n), push - O(1))

```c++
#include <bits/stdc++.h>

using namespace std;

class MyQueue {
	stack<int> input, output;
public:
	void push(int x) {
		input.push(x);
	}

	void pop() {
		peek();
		output.pop();
	}

	int peek() {
		if (output.size() == 0)
			while (input.size()) {
				output.push(input.top());
				input.pop();
			}
		return output.top();
	}
	bool empty() {
		return output.empty() && input.empty();
	}
};

int main() {
	MyQueue* obj = new MyQueue();

	obj->push(10);
	obj->push(20);
	obj->push(40);
	cout << obj->peek() << '\n';//10
	obj->pop();//10
	cout << obj->peek() << '\n';//20

}
```

##### 2. Costlier push (pop - O(1), push - O(n))

```c++
#include <bits/stdc++.h>

using namespace std;

class MyQueue {
	stack<int> input, output;
public:
	void push(int x) {
		if (input.empty())
			input.push(x); return;
		while (!input.empty()) {
			output.push(input.top());
			input.pop();
		}
		input.push(x);
		while (!output.empty()) {
			input.push(output.top());
			output.pop();
		}
		return;
	}

	void pop() {
		input.pop();
	}

	int peek() {

		return input.top();
	}

};

int main() {
	MyQueue* obj = new MyQueue();

	obj->push(10);
	obj->push(20);
	obj->push(40);
	cout << obj->peek() << '\n';//10
	obj->pop();//10
	obj->push(60);
	cout << obj->peek() << '\n';
	obj->pop();
	cout << obj->peek() << '\n';


}

```
맞게 했는데 런타임 오류가 나서 파이썬으로 구현했다.

```python
class MyQueue:

    def __init__(self):
        
        self.input1 = []
        self.output = []
        
    def push(self, x: int) -> None:
        
        self.input1.append(x)
        

    def pop(self) -> int:
        if not self.output:
            while self.input1:
                self.output.append(self.input1.pop())
        return self.output.pop()
        

    def peek(self) -> int:
        if not self.output:
            while self.input1:
                self.output.append(self.input1.pop())
        return self.output[-1]
        

    def empty(self) -> bool:
        return not (bool(self.input1) or bool(self.output))
        
```
